## 概述

libdnet是一个轻量级的网络编程库，提供了跨平台的底层网络协议封装结构。它允许开发者直接操作从数据链路层到应用层的各个协议层次，实现自定义的网络数据包构造、发送和接收。

### 核心设计原则

1. **统一抽象接口**：不同操作系统使用相同的API接口
2. **跨平台兼容**：支持Linux、BSD、Windows、Solaris等多个平台
3. **零拷贝设计**：直接操作原始数据包缓冲区
4. **模块化架构**：各协议层次独立实现，便于维护和扩展
5. **性能优先**：优化的数据包处理路径

### 支持的协议层次

```
┌─────────────────────────────────┐
│   应用层 (Application)          │
├─────────────────────────────────┤
│   传输层 (Transport)            │  TCP, UDP, SCTP
├─────────────────────────────────┤
│   网络层 (Network)              │  IP, IPv6, ICMP
├─────────────────────────────────┤
│   数据链路层 (Data Link)        │  Ethernet, ARP
├─────────────────────────────────┤
│   物理层 (Physical)            │  网卡驱动接口
└─────────────────────────────────┘
```

## 网络协议层次架构

### 1. 协议栈层次结构

libdnet按照OSI七层模型和TCP/IP四层模型实现了完整的协议栈封装：

#### 1.1 物理层抽象
- 通过操作系统网卡驱动接口访问
- 支持多种网络接口类型（以太网、回环、点对点等）

#### 1.2 数据链路层
- **Ethernet**：以太网帧封装和解析
- **ARP**：地址解析协议
- 硬件地址管理

#### 1.3 网络层
- **IP**：IPv4协议封装和路由
- **IPv6**：IPv6协议和扩展头处理
- **ICMP/ICMPv6**：互联网控制消息协议

#### 1.4 传输层
- **TCP**：传输控制协议
- **UDP**：用户数据报协议
- **SCTP**：流控制传输协议（CRC32C校验）

#### 1.2 跨平台实现

**Linux实现** (`src/eth-linux.c`):

```c
#include <linux/if_packet.h>
#include <sys/socket.h>
#include <net/if.h>

struct eth_handle {
    int fd;                     /* 原始套接字 */
    struct sockaddr_ll sll;     /* 链路层地址 */
    int iface_index;            /* 接口索引 */
};

eth_t *
eth_open(const char *device)
{
    eth_t *e;
    struct ifreq ifr;

    if ((e = calloc(1, sizeof(*e))) == NULL)
        return (NULL);
    
    /* 创建原始套接字 */
    if ((e->fd = socket(AF_PACKET, SOCK_RAW, htons(ETH_P_ALL))) < 0) {
        free(e);
        return (NULL);
    }
    
    /* 获取接口索引 */
    memset(&ifr, 0, sizeof(ifr));
    strlcpy(ifr.ifr_name, device, sizeof(ifr.ifr_name));
    if (ioctl(e->fd, SIOCGIFINDEX, &ifr) < 0) {
        close(e->fd);
        free(e);
        return (NULL);
    }
    e->iface_index = ifr.ifr_ifindex;
    
    /* 绑定到指定接口 */
    memset(&e->sll, 0, sizeof(e->sll));
    e->sll.sll_family = AF_PACKET;
    e->sll.sll_ifindex = e->iface_index;
    e->sll.sll_protocol = htons(ETH_P_ALL);
    
    if (bind(e->fd, (struct sockaddr *)&e->sll, sizeof(e->sll)) < 0) {
        close(e->fd);
        free(e);
        return (NULL);
    }
    
    return (e);
}

ssize_t
eth_send(eth_t *e, const void *buf, size_t len)
{
    /* 直接发送以太网帧 */
    return (write(e->fd, buf, len));
}

ssize_t
eth_recv(eth_t *e, void *buf, size_t len)
{
    /* 接收以太网帧 */
    return (read(e->fd, buf, len));
}
```

**BSD实现** (`src/eth-bsd.c`):

```c
#include <net/bpf.h>
#include <sys/ioctl.h>

struct eth_handle {
    int fd;                     /* BPF设备文件描述符 */
    int bufsize;                /* BPF缓冲区大小 */
};

eth_t *
eth_open(const char *device)
{
    eth_t *e;
    int i, n = 0;
    struct ifreq ifr;
    char dev[16];

    /* 查找可用的BPF设备 */
    for (i = 0; i < 256; i++) {
        snprintf(dev, sizeof(dev), "/dev/bpf%d", i);
        if ((e->fd = open(dev, O_RDWR)) >= 0) {
            n = i;
            break;
        }
    }
    if (e->fd < 0)
        return (NULL);
    
    /* 绑定到网络接口 */
    memset(&ifr, 0, sizeof(ifr));
    strlcpy(ifr.ifr_name, device, sizeof(ifr.ifr_name));
    if (ioctl(e->fd, BIOCSETIF, &ifr) < 0) {
        close(e->fd);
        free(e);
        return (NULL);
    }
    
    /* 设置立即模式 */
    n = 1;
    ioctl(e->fd, BIOCIMMEDIATE, &n);
    
    /* 设置缓冲区大小 */
    ioctl(e->fd, BIOCSBLEN, &e->bufsize);
    
    return (e);
}
```

**Windows实现** (`src/eth-win32.c`):

```c
#include <winsock2.h>
#include <pcap.h>

struct eth_handle {
    pcap_t *pcap;               /* WinPcap句柄 */
    char device[256];           /* 设备名称 */
};

eth_t *
eth_open(const char *device)
{
    eth_t *e;
    char errbuf[PCAP_ERRBUF_SIZE];

    if ((e = calloc(1, sizeof(*e))) == NULL)
        return (NULL);
    
    /* 使用WinPcap打开设备 */
    e->pcap = pcap_open_live(device, 65535, 1, 1000, errbuf);
    if (e->pcap == NULL) {
        free(e);
        return (NULL);
    }
    
    strlcpy(e->device, device, sizeof(e->device));
    
    return (e);
}

ssize_t
eth_send(eth_t *e, const void *buf, size_t len)
{
    if (pcap_sendpacket(e->pcap, (u_char *)buf, len) < 0)
        return (-1);
    return (len);
}

ssize_t
eth_recv(eth_t *e, void *buf, size_t len)
{
    struct pcap_pkthdr hdr;
    u_char *packet;
    
    packet = pcap_next(e->pcap, &hdr);
    if (packet == NULL)
        return (-1);
    
    if (hdr.caplen > len)
        return (-1);
    
    memcpy(buf, packet, hdr.caplen);
    return (hdr.caplen);
}
```

## ARP协议实现

### 1. ARP数据结构

**文件**: `include/dnet/arp.h`

```c
/* ARP头部固定长度 */
#define ARP_HDR_LEN          8

/* Ethernet/IP ARP消息长度 */
#define ARP_ETHIP_LEN        20

/* ARP头部 */
struct arp_hdr {
    uint16_t ar_hrd;         /* 硬件地址格式 */
    uint16_t ar_pro;         /* 协议地址格式 */
    uint8_t  ar_hln;         /* 硬件地址长度 */
    uint8_t  ar_pln;         /* 协议地址长度 */
    uint16_t ar_op;          /* 操作类型 */
} __attribute__((__packed__));

/* 硬件地址类型 */
#define ARP_HRD_ETH          0x0001    /* 以太网 */
#define ARP_HRD_IEEE802      0x0006    /* IEEE 802 */
#define ARP_HRD_IEEE80211    0x0321    /* IEEE 802.11 */

/* 协议地址类型 */
#define ARP_PRO_IP           0x0800    /* IP协议 */

/* ARP操作类型 */
#define ARP_OP_REQUEST       1         /* ARP请求 */
#define ARP_OP_REPLY         2         /* ARP响应 */

/* Ethernet/IP ARP消息 */
struct arp_ethip {
    eth_addr_t ar_sha;       /* 发送方硬件地址 */
    ip_addr_t  ar_spa;       /* 发送方协议地址 */
    eth_addr_t ar_tha;       /* 目标硬件地址 */
    ip_addr_t  ar_tpa;       /* 目标协议地址 */
} __attribute__((__packed__));

/* ARP缓存条目 */
struct arp_entry {
    struct addr arp_pa;       /* 协议地址 */
    struct addr arp_ha;       /* 硬件地址 */
};
```

### 2. 跨平台ARP实现

#### 2.1 BSD平台实现

**文件**: `src/arp-bsd.c`

BSD平台使用路由套接字（Routing Socket）来操作ARP缓存：

```c
struct arp_handle {
    int fd;                  /* 路由套接字 */
    int seq;                 /* 消息序列号 */
};

/* ARP消息结构 */
struct arpmsg {
    struct rt_msghdr rtm;    /* 路由消息头 */
    u_char          addrs[256];  /* 地址数据 */
};

arp_t *
arp_open(void)
{
    arp_t *arp;

    if ((arp = calloc(1, sizeof(*arp))) != NULL) {
        /* 创建路由套接字 */
        if ((arp->fd = socket(PF_ROUTE, SOCK_RAW, 0)) < 0)
            return (arp_close(arp));
    }
    return (arp);
}

int
arp_add(arp_t *arp, const struct arp_entry *entry)
{
    struct arpmsg msg;
    struct sockaddr_in *sin;
    struct sockaddr *sa;
    int index, type;

    /* 构造RTM_ADD消息 */
    sin = (struct sockaddr_in *)msg.addrs;
    sa = (struct sockaddr *)(sin + 1);

    /* 设置协议地址 */
    if (addr_ntos(&entry->arp_pa, (struct sockaddr *)sin) < 0)
        return (-1);

    /* 先查询获取接口信息 */
    memset(&msg.rtm, 0, sizeof(msg.rtm));
    msg.rtm.rtm_type = RTM_GET;
    msg.rtm.rtm_addrs = RTA_DST;
    msg.rtm.rtm_msglen = sizeof(msg.rtm) + sizeof(*sin);

    if (arp_msg(arp, &msg) < 0)
        return (-1);

    /* 获取接口索引和类型 */
    if (sa->sa_family != AF_LINK) {
        errno = EADDRNOTAVAIL;
        return (-1);
    }
    index = ((struct sockaddr_dl *)sa)->sdl_index;
    type = ((struct sockaddr_dl *)sa)->sdl_type;

    /* 设置ARP条目 */
    if (addr_ntos(&entry->arp_pa, (struct sockaddr *)sin) < 0 ||
        addr_ntos(&entry->arp_ha, sa) < 0)
        return (-1);

    ((struct sockaddr_dl *)sa)->sdl_index = index;
    ((struct sockaddr_dl *)sa)->sdl_type = type;

    /* 发送RTM_ADD消息添加ARP条目 */
    memset(&msg.rtm, 0, sizeof(msg.rtm));
    msg.rtm.rtm_type = RTM_ADD;
    msg.rtm.rtm_addrs = RTA_DST | RTA_GATEWAY;
    msg.rtm.rtm_inits = RTV_EXPIRE;
    msg.rtm.rtm_flags = RTF_HOST | RTF_STATIC;
    msg.rtm.rtm_msglen = sizeof(msg.rtm) + sizeof(*sin) + sizeof(*sa);

    return (arp_msg(arp, &msg));
}

int
arp_get(arp_t *arp, struct arp_entry *entry)
{
    struct arpmsg msg;
    struct sockaddr_in *sin;
    struct sockaddr *sa;

    sin = (struct sockaddr_in *)msg.addrs;
    sa = (struct sockaddr *)(sin + 1);

    /* 设置查询地址 */
    if (addr_ntos(&entry->arp_pa, (struct sockaddr *)sin) < 0)
        return (-1);

    /* 发送RTM_GET查询 */
    memset(&msg.rtm, 0, sizeof(msg.rtm));
    msg.rtm.rtm_type = RTM_GET;
    msg.rtm.rtm_addrs = RTA_DST;
    msg.rtm.rtm_flags = RTF_LLINFO;
    msg.rtm.rtm_msglen = sizeof(msg.rtm) + sizeof(*sin);

    if (arp_msg(arp, &msg) < 0)
        return (-1);

    /* 解析返回的硬件地址 */
    if (sin->sin_addr.s_addr != entry->arp_pa.addr_ip ||
        sa->sa_family != AF_LINK) {
        errno = ESRCH;
        return (-1);
    }
    
    return (addr_ston(sa, &entry->arp_ha));
}

int
arp_loop(arp_t *arp, arp_handler callback, void *arg)
{
    struct arp_entry entry;
    struct rt_msghdr *rtm;
    struct sockaddr_in *sin;
    struct sockaddr *sa;
    char *buf, *lim, *next;
    size_t len;
    int ret;
    
    /* 使用sysctl获取ARP缓存 */
    int mib[6] = { CTL_NET, PF_ROUTE, 0, AF_INET,
                   NET_RT_FLAGS, RTF_LLINFO };

    if (sysctl(mib, 6, NULL, &len, NULL, 0) < 0)
        return (-1);

    if ((buf = malloc(len)) == NULL)
        return (-1);

    if (sysctl(mib, 6, buf, &len, NULL, 0) < 0) {
        free(buf);
        return (-1);
    }
    
    lim = buf + len;
    ret = 0;

    /* 遍历所有ARP条目 */
    for (next = buf; next < lim; next += rtm->rtm_msglen) {
        rtm = (struct rt_msghdr *)next;
        sin = (struct sockaddr_in *)(rtm + 1);
        sa = (struct sockaddr *)(sin + 1);

        if (addr_ston((struct sockaddr *)sin, &entry.arp_pa) < 0 ||
            addr_ston(sa, &entry.arp_ha) < 0)
            continue;

        if ((ret = callback(&entry, arg)) != 0)
            break;
    }
    free(buf);

    return (ret);
}
```

#### 2.2 Linux平台实现

**文件**: `src/arp-ioctl.c`

Linux平台使用ioctl操作ARP表：

```c
struct arp_handle {
    int fd;                   /* 原始套接字 */
    intf_t *intf;             /* 接口句柄 */
};

arp_t *
arp_open(void)
{
    arp_t *a;

    if ((a = calloc(1, sizeof(*a))) != NULL) {
        /* 创建IP套接字 */
        if ((a->fd = socket(AF_INET, SOCK_DGRAM, 0)) < 0)
            return (arp_close(a));
        
        /* 打开接口句柄 */
        if ((a->intf = intf_open()) == NULL)
            return (arp_close(a));
    }
    return (a);
}

int
arp_add(arp_t *a, const struct arp_entry *entry)
{
    struct arpreq ar;

    memset(&ar, 0, sizeof(ar));

    /* 设置协议地址 */
    if (addr_ntos(&entry->arp_pa, &ar.arp_pa) < 0)
        return (-1);

    /* 设置硬件地址 */
    if (addr_ntos(&entry->arp_ha, &ar.arp_ha) < 0)
        return (-1);
    ar.arp_ha.sa_family = ARP_HRD_ETH;

    /* 查找接口 */
    if (intf_loop(a->intf, _arp_set_dev, &ar) != 1) {
        errno = ESRCH;
        return (-1);
    }
    
    /* 设置ARP标志 */
    ar.arp_flags = ATF_PERM | ATF_COM;

    /* 使用SIOCSARP添加ARP条目 */
    if (ioctl(a->fd, SIOCSARP, &ar) < 0)
        return (-1);

    return (0);
}

int
arp_get(arp_t *a, struct arp_entry *entry)
{
    struct arpreq ar;

    memset(&ar, 0, sizeof(ar));

    /* 设置查询地址 */
    if (addr_ntos(&entry->arp_pa, &ar.arp_pa) < 0)
        return (-1);

    /* 查找接口 */
    if (intf_loop(a->intf, _arp_set_dev, &ar) != 1) {
        errno = ESRCH;
        return (-1);
    }
    
    /* 使用SIOCGARP查询ARP条目 */
    if (ioctl(a->fd, SIOCGARP, &ar) < 0)
        return (-1);

    /* 检查是否完成 */
    if ((ar.arp_flags & ATF_COM) == 0) {
        errno = ESRCH;
        return (-1);
    }
    
    return (addr_ston(&ar.arp_ha, &entry->arp_ha));
}

int
arp_loop(arp_t *a, arp_handler callback, void *arg)
{
    FILE *fp;
    struct arp_entry entry;
    char buf[BUFSIZ], ipbuf[100], macbuf[100];
    int flags, ret;

    /* 从/proc/net/arp读取ARP表 */
    if ((fp = fopen("/proc/net/arp", "r")) == NULL)
        return (-1);

    ret = 0;
    while (fgets(buf, sizeof(buf), fp) != NULL) {
        /* 跳过标题行和未完成条目 */
        if (sscanf(buf, "%s 0x%x 0x%x %99s",
            ipbuf, &flags, &flags, macbuf) < 4 ||
            (flags & ATF_COM) == 0)
            continue;

        /* 解析IP和MAC地址 */
        if (addr_aton(ipbuf, &entry.arp_pa) == 0 &&
            addr_aton(macbuf, &entry.arp_ha) == 0) {
            if ((ret = callback(&entry, arg)) != 0)
                break;
        }
    }
    fclose(fp);

    return (ret);
}
```

```c
struct ip_handle {
    int fd;                  /* 原始套接字 */
};

ip_t *
ip_open(void)
{
    ip_t *i;
    int n;
    socklen_t len;

    if ((i = calloc(1, sizeof(*i))) == NULL)
        return (NULL);

    /* 创建IPV4原始套接字 */
    if ((i->fd = socket(AF_INET, SOCK_RAW, IPPROTO_RAW)) < 0)
        return (ip_close(i));
    
#ifdef IP_HDRINCL
    /* 设置IP_HDRINCL选项，由用户构造IP头 */
    n = 1;
    if (setsockopt(i->fd, IPPROTO_IP, IP_HDRINCL, &n, sizeof(n)) < 0)
        return (ip_close(i));
#endif
    
#ifdef SO_SNDBUF
    /* 优化发送缓冲区大小 */
    len = sizeof(n);
    if (getsockopt(i->fd, SOL_SOCKET, SO_SNDBUF, &n, &len) < 0)
        return (ip_close(i));

    for (n += 128; n < 1048576; n += 128) {
        if (setsockopt(i->fd, SOL_SOCKET, SO_SNDBUF, &n, len) < 0) {
            if (errno == ENOBUFS)
                break;
            return (ip_close(i));
        }
    }
#endif
    
#ifdef SO_BROADCAST
    /* 允许广播 */
    n = 1;
    if (setsockopt(i->fd, SOL_SOCKET, SO_BROADCAST, &n, sizeof(n)) < 0)
        return (ip_close(i));
#endif
    
    return (i);
}

ssize_t
ip_send(ip_t *i, const void *buf, size_t len)
{
    struct ip_hdr *ip;
    struct sockaddr_in sin;

    ip = (struct ip_hdr *)buf;

    /* 构造目标地址 */
    memset(&sin, 0, sizeof(sin));
#ifdef HAVE_SOCKADDR_SA_LEN       
    sin.sin_len = sizeof(sin);
#endif
    sin.sin_family = AF_INET;
    sin.sin_addr.s_addr = ip->ip_dst;
    
#ifdef HAVE_RAWIP_HOST_OFFLEN
    /* 某些平台需要处理字节序 */
    ip->ip_len = ntohs(ip->ip_len);
    ip->ip_off = ntohs(ip->ip_off);

    len = sendto(i->fd, buf, len, 0,
        (struct sockaddr *)&sin, sizeof(sin));
    
    ip->ip_len = htons(ip->ip_len);
    ip->ip_off = htons(ip->ip_off);

    return (len);
#else
    return (sendto(i->fd, buf, len, 0,
        (struct sockaddr *)&sin, sizeof(sin)));
#endif
}
```

#### 2.2 IPv6扩展头处理

**文件**: `src/ip6.c`

```c
/* 检查是否为IPv6扩展头 */
#define IP6_IS_EXT(n)  \
    ((n) == IP_PROTO_HOPOPTS || (n) == IP_PROTO_DSTOPTS || \
     (n) == IP_PROTO_ROUTING || (n) == IP_PROTO_FRAGMENT)

void
ip6_checksum(void *buf, size_t len)
{
    struct ip6_hdr *ip6 = (struct ip6_hdr *)buf;
    struct ip6_ext_hdr *ext;
    u_char *p, nxt;
    int i, sum;

    nxt = ip6->ip6_nxt;

    /* 跳过扩展头，找到上层协议 */
    for (i = IP6_HDR_LEN; IP6_IS_EXT(nxt); i += (ext->ext_len + 1) << 3) {
        if (i >= (int)len) return;
        ext = (struct ip6_ext_hdr *)((u_char *)buf + i);
        nxt = ext->ext_nxt;
    }
    
    p = (u_char *)buf + i;
    len -= i;

    /* 根据上层协议计算校验和 */
    if (nxt == IP_PROTO_TCP) {
        struct tcp_hdr *tcp = (struct tcp_hdr *)p;

        if (len >= TCP_HDR_LEN) {
            tcp->th_sum = 0;
            /* 伪头部 + TCP头部 + 数据 */
            sum = ip_cksum_add(tcp, len, 0) + htons(nxt + (u_short)len);
            sum = ip_cksum_add(&ip6->ip6_src, 32, sum);
            tcp->th_sum = ip_cksum_carry(sum);
        }
    } else if (nxt == IP_PROTO_UDP) {
        struct udp_hdr *udp = (struct udp_hdr *)p;

        if (len >= UDP_HDR_LEN) {
            udp->uh_sum = 0;
            sum = ip_cksum_add(udp, len, 0) + htons(nxt + (u_short)len);
            sum = ip_cksum_add(&ip6->ip6_src, 32, sum);
            /* UDP校验和为0的特殊处理 */
            if ((udp->uh_sum = ip_cksum_carry(sum)) == 0)
                udp->uh_sum = 0xffff;
        }
    } else if (nxt == IP_PROTO_ICMPV6) {
        struct icmp_hdr *icmp = (struct icmp_hdr *)p;

        if (len >= ICMP_HDR_LEN) {
            icmp->icmp_cksum = 0;
            sum = ip_cksum_add(icmp, len, 0) + htons(nxt + (u_short)len);
            sum = ip_cksum_add(&ip6->ip6_src, 32, sum);
            icmp->icmp_cksum = ip_cksum_carry(sum);
        }
    }
}
```

```c
#include <linux/rtnetlink.h>

struct route_handle {
    int fd;                  /* Netlink套接字 */
    unsigned int seq;        /* 消息序列号 */
};

route_t *
route_open(void)
{
    route_t *r;
    struct sockaddr_nl sa;

    if ((r = calloc(1, sizeof(*r))) == NULL)
        return (NULL);

    /* 创建Netlink套接字 */
    if ((r->fd = socket(AF_NETLINK, SOCK_RAW, NETLINK_ROUTE)) < 0) {
        free(r);
        return (NULL);
    }

    /* 绑定到Netlink */
    memset(&sa, 0, sizeof(sa));
    sa.nl_family = AF_NETLINK;
    sa.nl_pid = getpid();
    sa.nl_groups = RTMGRP_IPV4_ROUTE | RTMGRP_IPV6_ROUTE;

    if (bind(r->fd, (struct sockaddr *)&sa, sizeof(sa)) < 0) {
        close(r->fd);
        free(r);
        return (NULL);
    }

    return (r);
}

int
route_add(route_t *r, const struct route_entry *entry)
{
    struct {
        struct nlmsghdr nlh;
        struct rtmsg rt;
        char buf[256];
    } req;
    struct rtattr *rta;
    
    /* 构造Netlink消息 */
    memset(&req, 0, sizeof(req));
    
    req.nlh.nlmsg_len = NLMSG_LENGTH(sizeof(req.rt));
    req.nlh.nlmsg_type = RTM_NEWROUTE;
    req.nlh.nlmsg_flags = NLM_F_REQUEST | NLM_F_CREATE | NLM_F_EXCL;
    req.nlh.nlmsg_seq = ++r->seq;
    
    req.rt.rtm_family = AF_INET;
    req.rt.rtm_table = RT_TABLE_MAIN;
    req.rt.rtm_protocol = RTPROT_BOOT;
    req.rt.rtm_scope = RT_SCOPE_UNIVERSE;
    req.rt.rtm_type = RTN_UNICAST;
    req.rt.rtm_dst_len = entry->route_dst.addr_bits;
    
    /* 添加目的地址属性 */
    rta = (struct rtattr *)((char *)&req + NLMSG_ALIGN(req.nlh.nlmsg_len));
    rta->rta_type = RTA_DST;
    rta->rta_len = RTA_LENGTH(sizeof(struct in_addr));
    memcpy(RTA_DATA(rta), &entry->route_dst.addr_ip, sizeof(struct in_addr));
    req.nlh.nlmsg_len = NLMSG_ALIGN(req.nlh.nlmsg_len) + RTA_ALIGN(rta->rta_len);
    
    /* 添加网关属性 */
    rta = (struct rtattr *)((char *)&req + NLMSG_ALIGN(req.nlh.nlmsg_len));
    rta->rta_type = RTA_GATEWAY;
    rta->rta_len = RTA_LENGTH(sizeof(struct in_addr));
    memcpy(RTA_DATA(rta), &entry->route_gw.addr_ip, sizeof(struct in_addr));
    req.nlh.nlmsg_len = NLMSG_ALIGN(req.nlh.nlmsg_len) + RTA_ALIGN(rta->rta_len);
    
    /* 发送Netlink消息 */
    return (send(r->fd, &req, req.nlh.nlmsg_len, 0) < 0 ? -1 : 0);
}

int
route_loop(route_t *r, route_handler callback, void *arg)
{
    struct {
        struct nlmsghdr nlh;
        struct rtmsg rt;
        char buf[8192];
    } req, resp;
    struct rtattr *rta;
    struct route_entry entry;
    int len, done = 0;
    
    /* 构造查询请求 */
    memset(&req, 0, sizeof(req));
    req.nlh.nlmsg_len = NLMSG_LENGTH(sizeof(req.rt));
    req.nlh.nlmsg_type = RTM_GETROUTE;
    req.nlh.nlmsg_flags = NLM_F_REQUEST | NLM_F_DUMP;
    req.nlh.nlmsg_seq = ++r->seq;
    
    req.rt.rtm_family = AF_UNSPEC;
    
    /* 发送请求 */
    if (send(r->fd, &req, req.nlh.nlmsg_len, 0) < 0)
        return (-1);
    
    /* 接收并处理响应 */
    while (!done) {
        if ((len = recv(r->fd, &resp, sizeof(resp), 0)) < 0)
            return (-1);
        
        /* 解析Netlink消息 */
        for (struct nlmsghdr *nlh = &resp.nlh; 
             NLMSG_OK(nlh, len); 
             nlh = NLMSG_NEXT(nlh, len)) {
            
            if (nlh->nlmsg_type == NLMSG_DONE) {
                done = 1;
                break;
            }
            
            if (nlh->nlmsg_type == NLMSG_ERROR)
                return (-1);
            
            /* 解析路由属性 */
            memset(&entry, 0, sizeof(entry));
            entry.route_dst.addr_type = ADDR_TYPE_IP;
            entry.route_gw.addr_type = ADDR_TYPE_IP;
            
            rta = RTM_RTA(NLMSG_DATA(nlh));
            int rta_len = RTM_PAYLOAD(nlh);
            
            for (; RTA_OK(rta, rta_len); rta = RTA_NEXT(rta, rta_len)) {
                switch (rta->rta_type) {
                case RTA_DST:
                    entry.route_dst.addr_ip = 
                        *(uint32_t *)RTA_DATA(rta);
                    entry.route_dst.addr_bits = 
                        ((struct rtmsg *)NLMSG_DATA(nlh))->rtm_dst_len;
                    break;
                case RTA_GATEWAY:
                    entry.route_gw.addr_ip = 
                        *(uint32_t *)RTA_DATA(rta);
                    break;
                case RTA_OIF:
                    if_indextoname(*(int *)RTA_DATA(rta),
                        entry.intf_name);
                    break;
                }
            }
            
            if (callback(&entry, arg) != 0)
                return (0);
        }
    }
    
    return (0);
}
```

```c
struct intf_handle {
    int  fd;                   /* IPv4 socket */
    int  fd6;                  /* IPv6 socket */
    struct ifconf ifc;        /* IPv4接口配置 */
#ifdef SIOCGLIFCONF
    struct lifconf lifc;      /* IPv6接口配置 */
#endif
    u_char  ifcbuf[4192];     /* 接口配置缓冲区 */
};

intf_t *
intf_open(void)
{
    intf_t *intf;
    int one = 1;

    if ((intf = calloc(1, sizeof(*intf))) != NULL) {
        intf->fd = intf->fd6 = -1;

        /* 打开IPv4套接字 */
        if ((intf->fd = socket(AF_INET, SOCK_DGRAM, 0)) < 0)
            return (intf_close(intf));

        setsockopt(intf->fd, SOL_SOCKET, SO_BROADCAST,
            (const char *) &one, sizeof(one));

        /* 打开IPv6套接字 */
#if defined(SIOCGLIFCONF) || defined(SIOCGIFNETMASK_IN6) || defined(SIOCGIFNETMASK6)
        if ((intf->fd6 = socket(AF_INET6, SOCK_DGRAM, 0)) < 0) {
# ifdef EPROTONOSUPPORT
            if (errno != EPROTONOSUPPORT)
# endif
                return (intf_close(intf));
        }
#endif
    }
    return (intf);
}

static int
_intf_set_type(struct intf_entry *entry)
{
    if ((entry->intf_flags & INTF_FLAG_LOOPBACK) != 0)
        entry->intf_type = INTF_TYPE_LOOPBACK;
    else if ((entry->intf_flags & INTF_FLAG_BROADCAST) != 0)
        entry->intf_type = INTF_TYPE_ETH;
    else if ((entry->intf_flags & INTF_FLAG_POINTOPOINT) != 0)
        entry->intf_type = INTF_TYPE_TUN;
    else
        entry->intf_type = INTF_TYPE_OTHER;
}

static int
_intf_get_noalias(intf_t *intf, struct intf_entry *entry)
{
    struct ifreq ifr;

    /* 获取接口索引 */
    entry->intf_index = if_nametoindex(entry->intf_name);
    if (entry->intf_index == 0)
        return (-1);

    strlcpy(ifr.ifr_name, entry->intf_name, sizeof(ifr.ifr_name));

    /* 获取接口标志 */
    if (ioctl(intf->fd, SIOCGIFFLAGS, &ifr) < 0)
        return (-1);

    entry->intf_flags = intf_iff_to_flags(ifr.ifr_flags);
    _intf_set_type(entry);

    /* 获取接口MTU */
#ifdef SIOCGIFMTU
    if (ioctl(intf->fd, SIOCGIFMTU, &ifr) < 0)
#endif
        return (-1);
    entry->intf_mtu = ifr.ifr_mtu;

    entry->intf_addr.addr_type = entry->intf_dst_addr.addr_type =
        entry->intf_link_addr.addr_type = ADDR_TYPE_NONE;

    /* 获取主IP地址 */
    if (ioctl(intf->fd, SIOCGIFADDR, &ifr) == 0) {
        addr_ston(&ifr.ifr_addr, &entry->intf_addr);
        if (ioctl(intf->fd, SIOCGIFNETMASK, &ifr) < 0)
            return (-1);
        addr_stob(&ifr.ifr_addr, &entry->intf_addr.addr_bits);
    }
    
    /* 获取其他地址 */
    if (entry->intf_type == INTF_TYPE_TUN) {
        if (ioctl(intf->fd, SIOCGIFDSTADDR, &ifr) == 0) {
            if (addr_ston(&ifr.ifr_addr, &entry->intf_dst_addr) < 0)
                return (-1);
        }
    } else if (entry->intf_type == INTF_TYPE_ETH) {
        /* 获取MAC地址 */
#if defined(SIOCGIFHWADDR)
        if (ioctl(intf->fd, SIOCGIFHWADDR, &ifr) < 0)
            return (-1);
        if (addr_ston(&ifr.ifr_addr, &entry->intf_link_addr) < 0)
            return (-1);
#elif defined(SIOCRPHYSADDR)
        /* Tru64 */
        struct ifdevea *ifd = (struct ifdevea *)&ifr;
        if (ioctl(intf->fd, SIOCRPHYSADDR, ifd) < 0)
            return (-1);
        addr_pack(&entry->intf_link_addr, ADDR_TYPE_ETH, ETH_ADDR_BITS,
            ifd->current_pa, ETH_ADDR_LEN);
#else
        /* 使用eth接口获取MAC地址 */
        eth_t *eth;
        if ((eth = eth_open(entry->intf_name)) != NULL) {
            if (!eth_get(eth, &entry->intf_link_addr.addr_eth)) {
                entry->intf_link_addr.addr_type = ADDR_TYPE_ETH;
                entry->intf_link_addr.addr_bits = ETH_ADDR_BITS;
            }
            eth_close(eth);
        }
#endif
    }
    return (0);
}

int
intf_set(intf_t *intf, const struct intf_entry *entry)
{
    struct ifreq ifr;
    struct intf_entry *orig;
    struct addr bcast;
    u_char buf[BUFSIZ];

    orig = (struct intf_entry *)buf;
    orig->intf_len = sizeof(buf);
    strcpy(orig->intf_name, entry->intf_name);

    if (intf_get(intf, orig) < 0)
        return (-1);

    /* 删除现有别名 */
    if (_intf_delete_aliases(intf, orig) < 0)
        return (-1);

    /* 删除现有地址 */
    if (_intf_delete_addrs(intf, orig) < 0)
        return (-1);

    memset(&ifr, 0, sizeof(ifr));
    strlcpy(ifr.ifr_name, entry->intf_name, sizeof(ifr.ifr_name));

    /* 设置接口MTU */
    if (entry->intf_mtu != 0) {
        ifr.ifr_mtu = entry->intf_mtu;
#ifdef SIOCSIFMTU
        if (ioctl(intf->fd, SIOCSIFMTU, &ifr) < 0)
#endif
            return (-1);
    }
    
    /* 设置接口地址 */
    if (entry->intf_addr.addr_type == ADDR_TYPE_IP) {
        if (addr_ntos(&entry->intf_addr, &ifr.ifr_addr) < 0)
            return (-1);
        if (ioctl(intf->fd, SIOCSIFADDR, &ifr) < 0 && errno != EEXIST)
            return (-1);

        /* 设置子网掩码 */
        if (addr_btos(entry->intf_addr.addr_bits, &ifr.ifr_addr) == 0
#ifdef __linux__
            && entry->intf_addr.addr_ip != 0
#endif
        ) {
            if (ioctl(intf->fd, SIOCSIFNETMASK, &ifr) < 0)
                return (-1);
        }
        
        /* 设置广播地址 */
        if (addr_bcast(&entry->intf_addr, &bcast) == 0) {
            if (addr_ntos(&bcast, &ifr.ifr_broadaddr) == 0) {
                ioctl(intf->fd, SIOCSIFBRDADDR, &ifr);
            }
        }
    }
    
    /* 设置链路层地址 */
    if (entry->intf_link_addr.addr_type == ADDR_TYPE_ETH &&
        addr_cmp(&entry->intf_link_addr, &orig->intf_link_addr) != 0) {
#if defined(SIOCSIFHWADDR)
        if (addr_ntos(&entry->intf_link_addr, &ifr.ifr_hwaddr) < 0)
            return (-1);
        if (ioctl(intf->fd, SIOCSIFHWADDR, &ifr) < 0)
            return (-1);
#elif defined (SIOCSIFLLADDR)
        memcpy(ifr.ifr_addr.sa_data, &entry->intf_link_addr.addr_eth,
            ETH_ADDR_LEN);
        ifr.ifr_addr.sa_len = ETH_ADDR_LEN;
        if (ioctl(intf->fd, SIOCSIFLLADDR, &ifr) < 0)
            return (-1);
#else
        eth_t *eth;
        if ((eth = eth_open(entry->intf_name)) == NULL)
            return (-1);
        if (eth_set(eth, &entry->intf_link_addr.addr_eth) < 0) {
            eth_close(eth);
            return (-1);
        }
        eth_close(eth);
#endif
    }
    
    /* 设置点对点目的地址 */
    if (entry->intf_dst_addr.addr_type == ADDR_TYPE_IP) {
        if (addr_ntos(&entry->intf_dst_addr, &ifr.ifr_dstaddr) < 0)
            return (-1);
        if (ioctl(intf->fd, SIOCSIFDSTADDR, &ifr) < 0 &&
            errno != EEXIST)
            return (-1);
    }
    
    /* 添加别名 */
    if (_intf_add_aliases(intf, entry) < 0)
        return (-1);

    /* 设置接口标志 */
    if (ioctl(intf->fd, SIOCGIFFLAGS, &ifr) < 0)
        return (-1);

    ifr.ifr_flags = intf_flags_to_iff(entry->intf_flags, ifr.ifr_flags);

    if (ioctl(intf->fd, SIOCSIFFLAGS, &ifr) < 0)
        return (-1);

    return (0);
}
```

```c
/* Linux: 从/proc/net/dev获取接口列表 */
int
intf_loop(intf_t *intf, intf_handler callback, void *arg)
{
    FILE *fp;
    struct intf_entry *entry;
    char *p, buf[BUFSIZ], ebuf[BUFSIZ];
    int ret;

    entry = (struct intf_entry *)ebuf;

    if ((fp = fopen("/proc/net/dev", "r")) == NULL)
        return (-1);

    intf->ifc.ifc_buf = (caddr_t)intf->ifcbuf;
    intf->ifc.ifc_len = sizeof(intf->ifcbuf);

    if (ioctl(intf->fd, SIOCGIFCONF, &intf->ifc) < 0) {
        fclose(fp);
        return (-1);
    }

    ret = 0;
    while (fgets(buf, sizeof(buf), fp) != NULL) {
        if ((p = strchr(buf, ':')) == NULL)
            continue;
        *p = '\0';
        for (p = buf; *p == ' '; p++)
            ;

        memset(ebuf, 0, sizeof(ebuf));
        strlcpy(entry->intf_name, p, sizeof(entry->intf_name));
        strlcpy(entry->os_intf_name, p, sizeof(entry->os_intf_name));
        strlcpy(entry->pcap_intf_name, p, sizeof(entry->pcap_intf_name));
        
        /* 获取驱动信息 */
        intf_get_drv_info(intf, entry);
        
        entry->intf_len = sizeof(ebuf);

        if (_intf_get_noalias(intf, entry) < 0) {
            ret = -1;
            break;
        }
        if (_intf_get_aliases(intf, entry) < 0) {
            ret = -1;
            break;
        }
        if ((ret = (*callback)(entry, arg)) != 0)
            break;
    }
    if (ferror(fp))
        ret = -1;

    fclose(fp);

    return (ret);
}

/* 获取驱动信息（Linux） */
static int
intf_get_drv_info(intf_t *intf, struct intf_entry *entry)
{
    struct ifreq ifr;
    struct ethtool_drvinfo drvinfo;
    memset(&ifr, 0, sizeof(ifr));
    memset(&drvinfo, 0, sizeof(drvinfo));

    drvinfo.cmd = ETHTOOL_GDRVINFO;
    ifr.ifr_data = (caddr_t)&drvinfo;
    strlcpy(ifr.ifr_name, entry->intf_name, IFNAMSIZ);

    if (ioctl(intf->fd, SIOCETHTOOL, &ifr)) {
        return -1;
    }

    strncpy(entry->driver_name, drvinfo.driver, INTF_VERS_LEN);
    strncpy(entry->driver_vers, drvinfo.version, INTF_VERS_LEN);
    strncpy(entry->firmware_vers, drvinfo.fw_version, INTF_VERS_LEN);
    return 0;
}
```

```c
static void
fr_to_fwc(const struct fw_rule *fr, struct ip_fwchange *fwc)
{
    memset(fwc, 0, sizeof(*fwc));

    strlcpy(fwc->fwc_rule.ipfw.fw_vianame, fr->fw_device, IFNAMSIZ);
    
    /* 设置操作类型 */
    if (fr->fw_op == FW_OP_ALLOW)
        strlcpy(fwc->fwc_rule.label, IP_FW_LABEL_ACCEPT, 
            sizeof(fwc->fwc_rule.label));
    else
        strlcpy(fwc->fwc_rule.label, IP_FW_LABEL_BLOCK,
            sizeof(fwc->fwc_rule.label));

    /* 设置方向 */
    if (fr->fw_dir == FW_DIR_IN)
        strlcpy(fwc->fwc_label, IP_FW_LABEL_INPUT,
            sizeof(fwc->fwc_label));
    else
        strlcpy(fwc->fwc_label, IP_FW_LABEL_OUTPUT,
            sizeof(fwc->fwc_label));
    
    /* 设置协议和地址 */
    fwc->fwc_rule.ipfw.fw_proto = fr->fw_proto;
    fwc->fwc_rule.ipfw.fw_src.s_addr = fr->fw_src.addr_ip;
    fwc->fwc_rule.ipfw.fw_dst.s_addr = fr->fw_dst.addr_ip;
    addr_btom(fr->fw_src.addr_bits, &fwc->fwc_rule.ipfw.fw_smsk.s_addr,
        IP_ADDR_LEN);
    addr_btom(fr->fw_dst.addr_bits, &fwc->fwc_rule.ipfw.fw_dmsk.s_addr,
        IP_ADDR_LEN);

    /* 设置端口范围 */
    fwc->fwc_rule.ipfw.fw_spts[0] = fr->fw_sport[0];
    fwc->fwc_rule.ipfw.fw_spts[1] = fr->fw_sport[1];
    fwc->fwc_rule.ipfw.fw_dpts[0] = fr->fw_dport[0];
    fwc->fwc_rule.ipfw.fw_dpts[1] = fr->fw_dport[1];

int
fw_loop(fw_t *fw, fw_handler callback, void *arg)
{

    struct ip_fwchange fwc;
    struct fw_rule fr;
    char buf[BUFSIZ];

    int ret;

    if ((fp = fopen("/proc/net/ip_fwchains", "r")) == NULL)
        return (-1);

    while (fgets(buf, sizeof(buf), fp) != NULL) {
        if (sscanf(buf,
            "%8s %X/%X->%X/%X %s %hX %hX %hu %u %u %u %u "
            "%hu-%hu %hu-%hu A%X X%X %hX %u %hu %s\n",
            fwc.fwc_label,

            &fwc.fwc_rule.ipfw.fw_smsk.s_addr,
            &fwc.fwc_rule.ipfw.fw_dst.s_addr,
            &fwc.fwc_rule.ipfw.fw_dmsk.s_addr,
            fwc.fwc_rule.ipfw.fw_vianame,
            &fwc.fwc_rule.ipfw.fw_flg,
            &fwc.fwc_rule.ipfw.fw_invflg,
            &fwc.fwc_rule.ipfw.fw_proto,

            &fwc.fwc_rule.ipfw.fw_spts[0],
            &fwc.fwc_rule.ipfw.fw_spts[1],
            &fwc.fwc_rule.ipfw.fw_dpts[0],
            &fwc.fwc_rule.ipfw.fw_dpts[1],
            &tand, &txor,
            &fwc.fwc_rule.ipfw.fw_redirpt,
            &fwc.fwc_rule.ipfw.fw_mark,
            &fwc.fwc_rule.ipfw.fw_outputsize,
            fwc.fwc_rule.label) != 23)
            break;

        /* 转换字节序 */
        fwc.fwc_rule.ipfw.fw_src.s_addr =
            htonl(fwc.fwc_rule.ipfw.fw_src.s_addr);
        fwc.fwc_rule.ipfw.fw_dst.s_addr =
            htonl(fwc.fwc_rule.ipfw.fw_dst.s_addr);

            htonl(fwc.fwc_rule.ipfw.fw_dmsk.s_addr);

            return (ret);
```

```c
fr_to_ipfw(const struct fw_rule *fr, struct ip_fw *ipfw)
{
    int i;

    /* 设置方向 */
    if (fr->fw_dir == FW_DIR_IN) {

    } else {
        if (*fr->fw_device != '\0') {
            fr_to_ipfw_device(fr->fw_device,
                ipfw->fw_out_if.fu_via_if.name,
                &ipfw->fw_out_if.fu_via_if.unit);
            ipfw->fw_flg |= IP_FW_F_OIFNAME;
        }
        ipfw->fw_flg |= IP_FW_F_OUT;
    }

    if (fr->fw_op == FW_OP_ALLOW)
        ipfw->fw_flg |= IP_FW_F_ACCEPT;
    else
        ipfw->fw_flg |= IP_FW_F_DENY;
    
    /* 设置协议和地址 */
    ipfw->fw_prot = fr->fw_proto;
    ipfw->fw_src.s_addr = fr->fw_src.addr_ip;
    ipfw->fw_dst.s_addr = fr->fw_dst.addr_ip;
    addr_btom(fr->fw_src.addr_bits, &ipfw->fw_smsk.s_addr, IP_ADDR_LEN);
    addr_btom(fr->fw_dst.addr_bits, &ipfw->fw_dmsk.s_addr, IP_ADDR_LEN);

    /* 根据协议设置端口 */
    switch (fr->fw_proto) {
    case IP_PROTO_TCP:
    case IP_PROTO_UDP:
        i = 0;
        if (fr->fw_sport[0] != fr->fw_sport[1]) {
            ipfw->fw_flg |= IP_FW_F_SRNG;
            ipfw->fw_uar.fw_pts[i++] = fr->fw_sport[0];
            ipfw->fw_uar.fw_pts[i++] = fr->fw_sport[1];
            IP_FW_SETNSRCP(ipfw, 2);
        } else if (fr->fw_sport[0] > 0) {
            ipfw->fw_uar.fw_pts[i++] = fr->fw_sport[0];
            IP_FW_SETNSRCP(ipfw, 1);
        }
        if (fr->fw_dport[0] != fr->fw_dport[1]) {
            ipfw->fw_flg |= IP_FW_F_DRNG;
            ipfw->fw_uar.fw_pts[i++] = fr->fw_dport[0];
            ipfw->fw_uar.fw_pts[i++] = fr->fw_dport[1];
            IP_FW_SETNDSTP(ipfw, 2);
        } else if (fr->fw_dport[0] > 0) {
            ipfw->fw_uar.fw_pts[i++] = fr->fw_dport[0];
            IP_FW_SETNDSTP(ipfw, 1);
        }
        break;
    case IP_PROTO_ICMP:
        if (fr->fw_sport[1]) {
            ipfw->fw_uar.fw_icmptypes[fr->fw_sport[0] / 32] |=
                1 << (fr->fw_sport[0] % 32);
            ipfw->fw_flg |= IP_FW_F_ICMPBIT;
        }
        break;
    }

{
    fw_t *fw;

        if ((fw->fd = socket(AF_INET, SOCK_RAW, IPPROTO_IP)) < 0)
            return (fw_close(fw));

{
    struct ip_fw ipfw;
    
    fr_to_ipfw(rule, &ipfw);

    return (setsockopt(fw->fd, IPPROTO_IP, IP_FW_ADD,
        &ipfw, sizeof(ipfw)));
}

int
fw_delete(fw_t *fw, const struct fw_rule *rule)
{
    struct ip_fw *ipfw;
    struct fw_rule fr;
    int nbytes, nalloc, ret;
    u_char *buf, *new;

    nbytes = nalloc = sizeof(*ipfw);
    if ((buf = malloc(nbytes)) == NULL)
        return (-1);
    
    /* 获取所有规则 */
    while (nbytes >= nalloc) {
        nalloc = nalloc * 2 + 200;
        nbytes = nalloc;
        if ((new = realloc(buf, nbytes)) == NULL) {
            if (buf)
                free(buf);
            return (-1);
        }
        buf = new;
        if (getsockopt(fw->fd, IPPROTO_IP, IP_FW_GET,
                       buf, &nbytes) < 0) {
            free(buf);
            return (-1);
        }
    }
    ret = -1;

    /* 查找并删除匹配的规则 */
    for (ipfw = (struct ip_fw *)buf; ipfw->fw_number < 65535; ipfw++) {
        ipfw_to_fr(ipfw, &fr);
        if (fw_cmp(&fr, rule) == 0) {
            if (setsockopt(fw->fd, IPPROTO_IP, IP_FW_DEL,
                ipfw, sizeof(*ipfw)) < 0)
                ret = -2;
            else
                ret = 0;
            break;
        }
    }
    free(buf);
    
    if (ret < 0) {
        if (ret == -1)
            errno = ESRCH;
        return (-1);
    }
    return (0);
}
```

```c
/* 构造完整的以太网帧 */
int
build_eth_ip_packet(u_char *buf, size_t buflen,
    const eth_addr_t *dst, const eth_addr_t *src,
    const struct addr *src_ip, const struct addr *dst_ip,
    uint8_t proto, uint16_t sport, uint16_t dport,
    const u_char *data, size_t datalen)
{
    struct eth_hdr *eth;
    struct ip_hdr *ip;
    struct tcp_hdr *tcp;
    u_char *payload;
    size_t packet_len;
    
    /* 计算总长度 */
    packet_len = ETH_HDR_LEN + IP_HDR_LEN + TCP_HDR_LEN + datalen;
    if (packet_len > buflen)
        return (-1);
    
    /* 构造以太网头部 */
    eth = (struct eth_hdr *)buf;
    memcpy(&eth->eth_dst, dst, ETH_ADDR_LEN);
    memcpy(&eth->eth_src, src, ETH_ADDR_LEN);
    eth->eth_type = htons(ETH_TYPE_IP);
    
    /* 构造IP头部 */
    ip = (struct ip_hdr *)(buf + ETH_HDR_LEN);
    ip_pack_hdr(ip, IP_HDR_LEN / 4, 0,
        packet_len - ETH_HDR_LEN, ip_id(), 64,
        proto, src_ip->addr_ip, dst_ip->addr_ip);
    
    /* 构造TCP头部 */
    tcp = (struct tcp_hdr *)(buf + ETH_HDR_LEN + IP_HDR_LEN);
    tcp_pack_hdr(tcp, sport, dport,
        htonl(0), htonl(0), 0, TH_SYN,
        htons(8192), 0, 0, NULL);
    
    /* 复制负载数据 */
    payload = buf + ETH_HDR_LEN + IP_HDR_LEN + TCP_HDR_LEN;
    if (datalen > 0)
        memcpy(payload, data, datalen);
    
    /* 计算校验和 */
    tcp_checksum(ip, tcp, payload, datalen);
    
    return (packet_len);
}
```

### 4. SIMD优化（AVX2）

```c
#ifdef __AVX2__
#include <immintrin.h>

/* 使用AVX2加速IP校验和计算 */
uint16_t
ip_checksum_avx2(const void *buf, size_t len)
{
    uint32_t sum = 0;
    const uint16_t *ptr = (const uint16_t *)buf;
    size_t i;
    
    /* 32字节块对齐处理 */
    __m256i vsum = _mm256_setzero_si256();
    
    for (i = 0; i + 16 <= len / 2; i += 16) {
        __m256i vdata = _mm256_loadu_si256((__m256i const *)(ptr + i));
        vsum = _mm256_add_epi32(vsum, 
            _mm256_and_si256(vdata, _mm256_set1_epi32(0x0000FFFF)));
        vsum = _mm256_add_epi32(vsum, 
            _mm256_srli_epi32(vdata, 16));
    }
    
    /* 合并结果 */
    uint32_t tmp[8];
    _mm256_storeu_si256((__m256i *)tmp, vsum);
    for (i = 0; i < 8; i++)
        sum += tmp[i];
    
    /* 处理剩余部分 */
    while (len > 1) {
        sum += *ptr++;
        len -= 2;
    }
    
    return (~((sum >> 16) + (sum & 0xffff)) & 0xffff);
}
#endif
```

```c
resolve_mac_with_retry(const char *device, const char *target_ip,
    eth_addr_t *mac, int max_retries, int timeout_ms)

    struct timeval tv;
    
    for (retry = 0; retry < max_retries; retry++) {
        /* 发送ARP请求 */
        if (arp_request(device, target_ip) == 0) {

                return (0);
        }
        
        usleep(100000);  /* 等待100ms */
    }
    
    fprintf(stderr, "Failed to resolve MAC for %s after %d retries\n",
        target_ip, max_retries);
    return (-1);
}
```

```c
    double elapsed;

    timersub(&now, &stats->start_time, &diff);
    elapsed = diff.tv_sec + diff.tv_usec / 1000000.0;
    
    printf("\n=== Packet Statistics ===\n");
    printf("Sent:           %lu packets, %lu bytes\n", stats->sent, stats->sent_bytes);
    printf("Received:       %lu packets, %lu bytes\n", stats->recv, stats->recv_bytes);
    printf("Errors:         %lu\n", stats->errors);
    printf("Checksum errors: %lu\n", stats->checksum_errors);
    printf("Elapsed time:   %.2f seconds\n", elapsed);
    
    if (elapsed > 0) {
```

```c
/* 数据包接收线程 */
void *
packet_receiver(void *arg)
{
    eth_t *eth = (eth_t *)arg;
    u_char packet[65535];
    ssize_t len;
    
    while (1) {
        len = eth_recv(eth, packet, sizeof(packet));
        if (len < 0) {
            perror("eth_recv");
            break;
        }
        
        /* 处理接收到的数据包 */
        process_packet(packet, len);
    }
    
    return (NULL);
}

/* 处理数据包 */
void
process_packet(const u_char *packet, size_t len)
{
    struct eth_hdr *eth_hdr;
    struct ip_hdr *ip_hdr;
    struct tcp_hdr *tcp_hdr;
    uint16_t eth_type;
    
    /* 检查以太网头部 */
    if (len < ETH_HDR_LEN) {
        return;
    }
    
    eth_hdr = (struct eth_hdr *)packet;
    eth_type = ntohs(eth_hdr->eth_type);
    
    /* 只处理IPv4数据包 */
    if (eth_type != ETH_TYPE_IP) {
        return;
    }
    
    /* 检查IP头部 */
    if (len < ETH_HDR_LEN + IP_HDR_LEN) {
        return;
    }
    
    ip_hdr = (struct ip_hdr *)(packet + ETH_HDR_LEN);
    
    /* 验证IP校验和 */
    if (ip_checksum(ip_hdr, ip_hdr->ip_hl * 4) != 0) {
        printf("IP checksum error\n");
        return;
    }
    
    /* 处理TCP数据包 */
    if (ip_hdr->ip_p == IP_PROTO_TCP) {
        if (len < ETH_HDR_LEN + IP_HDR_LEN + TCP_HDR_LEN) {
            return;
        }
        
        tcp_hdr = (struct tcp_hdr *)(packet + ETH_HDR_LEN + 
                                      ip_hdr->ip_hl * 4);
        
        printf("TCP packet: %s:%d -> %s:%d\n",
            ip_ntoa(&ip_hdr->ip_src),
            ntohs(tcp_hdr->th_sport),
            ip_ntoa(&ip_hdr->ip_dst),
            ntohs(tcp_hdr->th_dport));
        
        /* 检查SYN+ACK响应 */
        if (tcp_hdr->th_flags & TH_SYN && tcp_hdr->th_flags & TH_ACK) {
            printf("  SYN+ACK received!\n");
        }
    }
}
```
